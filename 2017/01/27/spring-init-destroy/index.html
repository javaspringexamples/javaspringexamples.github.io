Example Spring 4 – Init Destroy methods with XML-based configuration

TOPICS:destroyinitspringxml

Posted By: admin janvier 27, 2017


This example presents the basic concept of the Init and Destroy of a bean by using Spring Annotation.

The technologies used are :

  • Spring 4.3.4
  • Junit 4.12
  • JDK 1.8
  • Maven 3.3.9

You can convert this example to an eclipse project by going to folder where is the pom.xml is, and use the following command :

mvn eclipse:eclipse

Here the project files :

Example Spring 4 – Init Destroy by XML

package com.codercocoon.springcoreInitDestroyXml;

import javax.annotation.PostConstruct;
import javax.annotation.PreDestroy;

import org.springframework.stereotype.Component;

/**
 * Example class : this is the example class containing methods which will be
 * invoked in the bean loading into and unloading from the container.
 * 
 * @author contact@codercocoon.com
 *
 */
public class Example {

	private boolean init;
	private boolean destroy;

	private String title = "I'm an example !";

	public String getTitle() {
		return title;
	}

	/**
	 * init() method will be invoked when the bean is loaded in the container,
	 * it set the attribute "init" at True when invoked.
	 * 
	 */
	public void init() {
		init = true;
	}

	/**
	 * destroy() method will be invoked when the bean is unloaded from the
	 * container, it set the attribute "destroy" at True when invoked.
	 * 
	 */
	public void destroy() {
		destroy = true;
	}

	/**
	 * Getter of the init attribute.
	 * 
	 */
	public boolean isInit() {
		return init;
	}

	/**
	 * Getter of the destroy attribute.
	 * 
	 */
	public boolean isDestroy() {
		return destroy;
	}
}

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans-4.3.xsd http://www.springframework.org/schema/context http://www.springframework.org/schema/context/spring-context-4.3.xsd">

	<!-- Example bean with parameterized init- and destroy method -->
	<bean id="example" class="com.codercocoon.springcoreInitDestroyXml.Example" init-method="init" destroy-method="destroy" />
</beans>

package com.codercocoon.test.springcoreInitDestroyXml;

import org.junit.Assert;
import org.junit.Test;
import org.junit.runner.RunWith;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.ApplicationContext;
import org.springframework.context.ConfigurableApplicationContext;
import org.springframework.test.context.ContextConfiguration;
import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;

import com.codercocoon.springcoreInitDestroyXml.Example;

/**
 * Test class instead XML-based config.
 * 
 * @author contact@codercocoon.com
 *
 */

@RunWith(SpringJUnit4ClassRunner.class)
@ContextConfiguration("classpath:applicationContext.xml")
public class ApplicationXMLConfigTest {

	@Autowired
	private ApplicationContext contextXMLConfig;

	/**
	 * This test case checks if the init-method and destroy method are invoked
	 * in loading and unloading the bean into and from the container.
	 *
	 */
	@Test
	public void testExample() {
		Example example = contextXMLConfig.getBean(Example.class);
		Assert.assertTrue(example.isInit());
		Assert.assertFalse(example.isDestroy());

		((ConfigurableApplicationContext) contextXMLConfig).close();
		Assert.assertTrue(example.isDestroy());
	}
}
