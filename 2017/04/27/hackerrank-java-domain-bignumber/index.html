Challenge : Java BigInteger [Easy]

/**
 * Hackerrank : Java domain > BigNumber > Java BigInteger
 *
 * https://www.hackerrank.com/challenges/java-biginteger/problem
 *
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		BigInteger a = sc.nextBigInteger();
		BigInteger b = sc.nextBigInteger();

		System.out.println(a.add(b));
		System.out.println(a.multiply(b));
	}
}

Challenge : Java BigDecimal [Medium]

/**
 * Hackerrank : Java domain > BigNumber > Java BigDecimal
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		// Input
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		String[] s = new String[n + 2];
		for (int i = 0; i < n; i++) {
			s[i] = sc.next();
		}
		sc.close();
	        Arrays.sort(s, new Comparator<String>() {
				@Override
				public int compare(String s1, String s2) {
					if (s1 != null && s2 != null) {
						BigDecimal b1 = new BigDecimal(s1);
						BigDecimal b2 = new BigDecimal(s2);
						return b2.compareTo(b1);
					} else {
						return 0;
					}
				}
			});
		// Output
		for (int i = 0; i < n - 2; i++) {
			System.out.println(s[i]);
		}
	}
}

Challenge : Java Primality Test [Easy]

/**
 * Hackerrank : Java domain > BigNumber > Java Primality Test
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		BigInteger n = in.nextBigInteger();
		in.close();
		
		System.out.println((n.isProbablePrime(1) ? "" : "not ") + "prime");
	}
}
