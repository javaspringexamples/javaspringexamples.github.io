Challenge : Java BigDecimal [Medium]

/**
 * Hackerrank : Java domain > BigNumber > Java BigDecimal
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		// Input
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		String[] s = new String[n + 2];
		for (int i = 0; i < n; i++) {
			s[i] = sc.next();
		}
		sc.close();
	        Arrays.sort(s, new Comparator<String>() {
				@Override
				public int compare(String s1, String s2) {
					if (s1 != null && s2 != null) {
						BigDecimal b1 = new BigDecimal(s1);
						BigDecimal b2 = new BigDecimal(s2);
						return b2.compareTo(b1);
					} else {
						return 0;
					}
				}
			});
		// Output
		for (int i = 0; i < n - 2; i++) {
			System.out.println(s[i]);
		}
	}
}

 * Hackerrank : Java domain > Strings > Java String Compare

public class Solution {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		String s = in.next();
		int n = in.nextInt();
		String min = s.substring(0, n);
		String max = s.substring(0, n);
		
		for (int i = 1; i <= s.length()-n; i++) {
			String tmp = s.substring(i, i + n);
			System.out.println(tmp);
			if (min.compareTo(tmp) > 0)
				min = tmp;
			if (max.compareTo(tmp) < 0)
				max = tmp;
		}
		System.out.println(min);
		System.out.println(max);
	}
}

Challenge : Java Anagrams [Easy]

/**
 * Hackerrank : Java domain > Strings > Java Anagrams
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	static boolean isAnagram(String a, String b) {
		boolean anagram = false;

		if (a.length() != b.length())
			return anagram;
		else {
			a = a.toLowerCase();
			b = b.toLowerCase();
			char[] a_tab = a.toCharArray();
			char[] b_tab = b.toCharArray();
			Arrays.sort(a_tab);
			Arrays.sort(b_tab);
			anagram = Arrays.equals(a_tab, b_tab);
		}

		return anagram;
	}
}

Challenge : Java Regex 2 – Duplicate Words [Easy]

/**
 * Hackerrank : Java domain > Strings > Java Regex 2 - Duplicate Words
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		String regex = "(\\b\\w+\\b)(\\s*\\1\\b)+";
		Pattern p = Pattern.compile(regex, Pattern.CASE_INSENSITIVE);

		Scanner in = new Scanner(System.in);
		int numSentences = Integer.parseInt(in.nextLine());

		while (numSentences-- > 0) {
			String input = in.nextLine();
			Matcher m = p.matcher(input);
			// Check for subsequences of input that match the compiled pattern
			while (m.find()) {
				input = input.replaceAll(m.group(), m.group(1));
			}
			// Prints the modified sentence.
			System.out.println(input);
		}
		in.close();
	}
}

Challenge : Tag Content Extractor [Medium]

/**
 * Hackerrank : Java domain > Strings > Tag Content Extractor
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		int testCases = Integer.parseInt(in.nextLine());
		while (testCases > 0) {
			String l = in.nextLine();
			// String pattern = Complete this line in the editable area below
			String pattern = "<(.+)>([^<>]+)</\\1>";

			Pattern r = Pattern.compile(pattern);
			Matcher m = r.matcher(l);

			boolean find = false;
			while (m.find()) {
				find = true;
				System.out.println(m.group(1));
				System.out.println(m.group(2));
			}
			if (!find)
				System.out.println("None");
			find = false;

			testCases--;
		}
	}
}

/**
 * Hackerrank : Java domain > Inroduction > Java Datatypes
 * Java has 8 primitive data types; char, boolean, byte, short, int, long,
 * float, and double. For this exercise, we'll work with the primitives used to
 * hold integer values (byte, short, int, and long).
 * 
 * @author contact@codercocoon.com
 *
 */
public class Solution {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int t = sc.nextInt();
		
		for (int i = 0; i < t; i++) {
			try {
				long x = sc.nextLong();
				System.out.println(x + " can be fitted in:");
				if (x >= Byte.MIN_VALUE && x <= Byte.MAX_VALUE)
					System.out.println("* byte");
				if (x >= Short.MIN_VALUE && x <= Short.MAX_VALUE)
					System.out.println("* short");
				if (x >= Integer.MIN_VALUE && x <= Integer.MAX_VALUE)
					System.out.println("* int");
				if (x >= Long.MIN_VALUE && x <= Long.MAX_VALUE)
					System.out.println("* long");
			} catch (Exception e) {
				System.out.println(sc.next() + " can't be fitted anywhere.");
			}
		}
	}
}
